Represents a party on Vega, could be an ethereum wallet address in the future

```graphql
type Party {
  id: ID!
  orders(
  skip: Int
  first: Int
  last: Int
): [Order!]
  ordersConnection(
  pagination: Pagination
): OrderConnection!
  trades(
  marketId: ID
  skip: Int
  first: Int
  last: Int
): [Trade!]
  tradesConnection(
  marketId: ID
  pagination: Pagination
): TradeConnection!
  accounts(
  marketId: ID
  asset: String
  type: AccountType
): [Account!]
  positions: [Position!]
  positionsConnection(
  market: ID
  pagination: Pagination
): PositionConnection!
  margins(
  marketId: ID
): [MarginLevels!]
  marginsConnection(
  marketId: ID
  pagination: Pagination
): MarginConnection!
  proposals(
  inState: ProposalState
): [Proposal]
  votes: [ProposalVote]
  votesConnection(
  pagination: Pagination
): ProposalVoteConnection!
  withdrawals: [Withdrawal!]
  withdrawalsConnection: WithdrawalsConnection!
  deposits: [Deposit!]
  depositsConnection: DepositsConnection!
  liquidityProvisions(
  market: String
  reference: String
): [LiquidityProvision!]
  liquidityProvisionsConnection(
  market: String
  reference: String
  pagination: Pagination
): LiquidityProvisionsConnection!
  delegations(
  nodeId: String
  skip: Int
  first: Int
  last: Int
): [Delegation!]
  stake: PartyStake!
  rewards(
  asset: String
  skip: Int
  first: Int
  last: Int
): [Reward]
  rewardsConnection(
  asset: String
  pagination: Pagination
): RewardsConnection
  rewardSummaries(
  asset: String
): [RewardSummary]
  rewardDetails: [RewardPerAssetDetail]
}
```
