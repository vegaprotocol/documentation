Represents a party on Vega, could be an ethereum wallet address in the future

```graphql
type Party {
  id: ID!
  orders(
  skip: Int
  first: Int
  last: Int
): [Order!]
  ordersConnection(
  dateRange: DateRange
  pagination: Pagination
): OrderConnection
  trades(
  marketId: ID
  skip: Int
  first: Int
  last: Int
): [Trade!]
  tradesConnection(
  marketId: ID
  dataRange: DateRange
  pagination: Pagination
): TradeConnection
  accounts(
  marketId: ID
  assetId: ID
  type: AccountType
): [Account!]
  accountsConnection(
  marketId: ID
  assetId: ID
  type: AccountType
  pagination: Pagination
): AccountsConnection
  positions: [Position!]
  positionsConnection(
  market: ID
  pagination: Pagination
): PositionConnection
  margins(
  marketId: ID
): [MarginLevels!]
  marginsConnection(
  marketId: ID
  pagination: Pagination
): MarginConnection
  proposals(
  inState: ProposalState
): [Proposal]
  proposalsConnection(
  proposalType: ProposalType
  inState: ProposalState
  pagination: Pagination
): ProposalsConnection
  votes: [ProposalVote]
  votesConnection(
  pagination: Pagination
): ProposalVoteConnection
  withdrawals: [Withdrawal!]
  withdrawalsConnection(
  dateRange: DateRange
  pagination: Pagination
): WithdrawalsConnection
  deposits: [Deposit!]
  depositsConnection(
  dateRange: DateRange
  pagination: Pagination
): DepositsConnection
  liquidityProvisions(
  market: String
  reference: String
): [LiquidityProvision!]
  liquidityProvisionsConnection(
  marketId: ID
  reference: String
  pagination: Pagination
): LiquidityProvisionsConnection
  delegations(
  nodeId: ID
  skip: Int
  first: Int
  last: Int
): [Delegation!]
  delegationsConnection(
  nodeId: ID
  pagination: Pagination
): DelegationsConnection
  stake: PartyStake!
  stakingSummary(
  pagination: Pagination
): StakingSummary!
  rewards(
  assetId: ID
  skip: Int
  first: Int
  last: Int
): [Reward]
  rewardsConnection(
  assetId: ID
  pagination: Pagination
): RewardsConnection
  rewardSummaries(
  assetId: ID
): [RewardSummary]
  rewardDetails: [RewardPerAssetDetail]
  transfersConnection(
  direction: TransferDirection
  pagination: Pagination
): TransferConnection
}
```
