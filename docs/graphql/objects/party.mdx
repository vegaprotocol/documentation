Represents a party on Vega, could be an ethereum wallet address in the future

```graphql
type Party {
  id: ID!
  orders(
  skip: Int
  first: Int
  last: Int
): [Order!]
  ordersPaged(
  pagination: Pagination
): OrderConnection!
  trades(
  marketId: ID
  skip: Int
  first: Int
  last: Int
): [Trade!]
  tradesPaged(
  marketId: ID
  pagination: Pagination
): TradeConnection!
  accounts(
  marketId: ID
  asset: String
  type: AccountType
): [Account!]
  positions: [Position!]
  margins(
  marketId: ID
): [MarginLevels!]
  marginsPaged(
  marketId: ID
  pagination: Pagination
): MarginConnection!
  proposals(
  inState: ProposalState
): [Proposal]
  votes: [ProposalVote]
  withdrawals: [Withdrawal!]
  deposits: [Deposit!]
  liquidityProvisions(
  market: String
  reference: String
): [LiquidityProvision!]
  delegations(
  nodeId: String
  skip: Int
  first: Int
  last: Int
): [Delegation!]
  stake: PartyStake!
  rewards(
  asset: String
  skip: Int
  first: Int
  last: Int
): [Reward]
  rewardsConnection(
  asset: String
  pagination: Pagination
): RewardsConnection
  rewardSummaries(
  asset: String
): [RewardSummary]
  rewardDetails: [RewardPerAssetDetail]
}
```
